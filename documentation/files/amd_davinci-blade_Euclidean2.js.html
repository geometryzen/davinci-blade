            var b0 = b[0];
            var b1 = b[1];
            var b2 = b[2];
            var b3 = b[3];
            var x0 = rcoE2(a0, a1, a2, a3, b0, b1, b2, b3, 0);
            var x1 = rcoE2(a0, a1, a2, a3, b0, b1, b2, b3, 1);
            var x2 = rcoE2(a0, a1, a2, a3, b0, b1, b2, b3, 2);
            var x3 = rcoE2(a0, a1, a2, a3, b0, b1, b2, b3, 3);
            return [x0, x1, x2, x3];
        };
